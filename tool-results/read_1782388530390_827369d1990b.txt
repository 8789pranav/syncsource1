   763→  const exitedCases = EXIT_CASES.filter(c => c.exitStatus === "Exited").length
   764→  const onHoldCases = EXIT_CASES.filter(c => c.exitStatus === "On Hold").length
   765→  const pendingResignations = RESIGNATION_REQUESTS.filter(r => r.status === "Pending Manager Approval" || r.status === "Pending HR Approval").length
   766→  const pendingManagerApproval = RESIGNATION_REQUESTS.filter(r => r.status === "Pending Manager Approval").length
   767→  const pendingHrApproval = RESIGNATION_REQUESTS.filter(r => r.status === "Pending HR Approval").length
   768→  const clearancePending = EXIT_CASES.filter(c => c.clearanceStatus === "Pending" || c.clearanceStatus === "In Progress").length
   769→  const clearanceOverdue = EXIT_CASES.filter(c => c.clearanceStatus === "Overdue").length
   770→  const assetPending = EXIT_CASES.filter(c => c.assetStatus === "Pending" || c.assetStatus === "Partial").length
   771→  const itPending = EXIT_CASES.filter(c => c.itAccessStatus === "Pending" || c.itAccessStatus === "Scheduled").length
   772→  const fnfPending = EXIT_CASES.filter(c => c.fnfStatus === "Not Started" || c.fnfStatus === "Draft" || c.fnfStatus === "Inputs Pending" || c.fnfStatus === "Calculated" || c.fnfStatus === "Under Review").length
   773→  const letterPending = EXIT_CASES.filter(c => c.letterStatus === "Not Started" || c.letterStatus === "Pending").length
   774→  const lwdToday = EXIT_CASES.filter(c => c.approvedLwd === new Date().toISOString().slice(0, 10)).length
   775→  const noticePeriod = EXIT_CASES.filter(c => c.currentStageId === "s5").length
   776→  const exitedThisMonth = EXIT_CASES.filter(c => c.exitStatus === "Exited" && c.actualLwd && new Date(c.actualLwd).getMonth() === now.getMonth()).length
   777→  const withdrawn = EXIT_CASES.filter(c => c.exitStatus === "Withdrawn").length
   778→  const terminated = EXIT_CASES.filter(c => c.exitType === "Involuntary Termination").length
   779→  const exitInterviewPending = EXIT_CASES.filter(c => c.currentStageId === "s6" || c.currentStageId === "s9").length
   780→  const highRisk = EXIT_CASES.filter(c => c.riskFlag === "high").length
   781→
   782→  return {
